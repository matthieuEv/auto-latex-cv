    }

%-----------PROJECTS-----------
\section{\color{airforceblue}PROJETS}    
\resumeItemListStart
    \vspace{0.5pt}
      {% for proj in projects %}
        \vspace{0.5pt}
        {% if proj.Link == '' %}
          \resumeItem{\normalsize{\textbf{ {{ proj.Name }} }, {{ proj.Description }} } }
        {% else %}
          \resumeItem{\normalsize{\textbf{ {{ proj.Name }} }, {{ proj.Description }} } \href{ {{ proj.Link }} }{\color{blue}\underline{GitHub}}}
        {% endif %}
      {% endfor %}
    \vspace{5pt}
\resumeItemListEnd  

\vspace{-12pt}

%-----------EXPERIENCE-----------
\section{\color{airforceblue}EXPÉRIENCES}
 \resumeSubHeadingListStart
  {% for exp in experience %}
    \resumeSubheading
      { {{ exp.Title }} }{ {{ exp.Company }} }{ {{ exp.Location }} }{ {{ exp.Start }} , {{ exp.End }} }
      \resumeItemListStart
      {% for desc in exp.Description %}
      \resumeItem{\normalsize{ {{ desc }} } }
      {% endfor %}
      \null\hfill\raisebox{30pt}[0pt][0pt]{\includegraphics[width=30pt, height=30pt]{ {{ exp.image }} }}

  {% endfor %}
 \resumeSubHeadingListEnd
\vspace{-20pt}

  {% endfor %}
